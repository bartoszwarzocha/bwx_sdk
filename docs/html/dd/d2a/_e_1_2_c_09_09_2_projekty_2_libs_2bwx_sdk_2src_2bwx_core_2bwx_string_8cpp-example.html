Splits a string into parts using a specified separator.

Splits a string into parts using a specified separator.This function splits the given string based on the provided separator and returns the result as a wxArrayString.

Parameters
strThe string to be split.
separatorThe separator that defines the split points.
Returns
wxArrayString containing the parts resulting from the split.
wxString text = "one,two,three";
wxString sep = ",";
wxArrayString result = bwxSimpleExplode(text, sep);
// result contains ["one", "two", "three"]

wxArrayString bwxSimpleExplode(const wxString& str, const wxString& separator)
{
wxArrayString a;
a.Clear();
wxString s = str + separator; // Append separator at the end
int l = s.Length();
wxString temp;
for (int i = 0; i < l; ++i)
{
if (s[i] != separator[0]) // Compare directly with the first character of the separator
{
temp += s[i];
}
else
{
if (!temp.IsEmpty())
{
a.Add(temp);
temp.Clear();
}
}
}
return a;
}
